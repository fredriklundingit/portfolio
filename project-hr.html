For this project, my focus lay on iterating the rendering pipeline, and most importantly, integrating the editor to use the same render pipeline as ingame, since our level designers and procedural artists had to start the game each time they wanted to see how it looked in game during our previous project.

Render Pipeline Refactor

I was responsible for a full refactor of the render pipeline, introducing a command pattern with render data containing commands for each pass (GBuffer pass, shadow pass etc). This made it easier to add features and also brought the editor up to speed, level designers could now see changes without launching the game. New features added this project: spotlights, SSAO, line lights (shout out to my classmate Mattias Strand for letting me implement his version) and a player flashlight.

// The function called by both ingame and editor in the end:
void RenderManager::ExecuteRenderPipeline(const NCE::RenderData& renderData, Tga::DepthBuffer* aDepthBuffer, bool aRenderToBackBuffer)
{
    auto& engine = *Engine::GetInstance();
    auto& graphicsEngine = engine.GetGraphicsEngine();
    auto& graphicsStateStack = graphicsEngine.GetGraphicsStateStack();
    Tga::DepthBuffer* depthBuffer = aDepthBuffer ? aDepthBuffer : Tga::DX11::DepthBuffer;
    if (!renderData.camera || !renderData.directionalLight || !renderData.ambientLight)
    {
        ERROR_PRINT("RenderData missing required camera or lights!");
        return;
    }
    // ***** DEFAULT VALUES *****
    graphicsStateStack.SetPostProcessingValues(myRenderResources.postProcessingValues);
    graphicsStateStack.SetRenderMode(myRenderMode);
    graphicsStateStack.SetCamera(*renderData.camera);
    graphicsStateStack.SetSamplerState(Tga::SamplerFilter::Trilinear, Tga::SamplerAddressMode::Wrap);
    graphicsStateStack.SetRasterizerState(Tga::RasterizerState::BackfaceCulling);
    graphicsStateStack.SetDepthStencilState(Tga::DepthStencilState::WriteLessOrEqual);
    graphicsStateStack.SetBlendState(Tga::BlendState::Disabled);
    graphicsStateStack.SetOrbValue(renderData.flashlightBatteryNormalized);
    AI::PollingStation3D::Get().SetCameraPtr(renderData.camera);

    RenderFlashlightShadowPass(renderData, graphicsStateStack);
    RenderGBufferPass(renderData, graphicsStateStack, depthBuffer);
    RenderSSAOPass(graphicsStateStack, graphicsEngine);
    RenderDecalPass(renderData, graphicsStateStack, depthBuffer);
    RenderLightPass(renderData, graphicsStateStack);
    RenderTransparentPass(renderData, graphicsStateStack, depthBuffer);
    RenderBloomPass(graphicsStateStack);

    if (aRenderToBackBuffer)
    {
        RenderPostProcessing(renderData, graphicsStateStack);
    }
#ifndef _RETAIL
    if (myDrawColliders || myDrawEnemyAggroRange || myDrawEnemyAlertRadius || myDrawEnemyPatrolRadius)
    {
        RenderDebugPass(renderData, graphicsStateStack, depthBuffer);
    }
#endif
}
// A render command
struct RenderCommand
{
    RenderCommandType type = RenderCommandType::Mesh;
    // === MESH DATA ===
    const Tga::Model* model = nullptr;
    int meshIndex = 0;
    Tga::Matrix4x4f transform;
    const Tga::TextureResource* textures[4] = { nullptr };
    const Tga::ModelShader* shader = nullptr;
    const Tga::Matrix4x4f* boneTransforms = nullptr;
    // === SPRITE DATA ===
    const Tga::SpriteSharedData* spriteSharedData = nullptr;
    const Tga::SpriteShader* spriteShaderOverride = nullptr;
    Tga::Sprite3DInstanceData spriteInstance;
    Tga::Sprite2DInstanceData sprite2DInstance;
    Tga::Vector4f   uiShaderParam = { 0.f, 0.f, 0.f, 0.f };
    bool            hasUIShaderParam = false;
    // === RENDER STATES ===
    Tga::RasterizerState rasterizerState = Tga::RasterizerState::BackfaceCulling;
    Tga::BlendState blendState = Tga::BlendState::Disabled;
    Tga::DepthStencilState depthStencilState = Tga::DepthStencilState::WriteLess;
    Tga::SamplerFilter samplerFilter = Tga::SamplerFilter::Trilinear;
    Tga::SamplerAddressMode samplerAddressMode = Tga::SamplerAddressMode::Wrap;
    // === OPTIONAL EXTRAS ===
    EffectMaterial* effectMaterial = nullptr;
    uint32_t texturesEnabled[4] = { false, false, false, false };
    float startUV[2] = { 0.f, 0.f };
    float endUV[2] = { 1.f, 1.f };
    // === VFX DATA ===
    bool isVFX = false;
    VFXInstanceData vfxInstanceData;
    // === SORTING/FILTERING ===
    float sortKey = 0.f;
    bool castsShadows = true;
    bool shouldFade = false;
    // === LAYER ===
    Tga::LayerType layer = Tga::LayerType::Default;
};
Editor Integration

Earlier the editor used its own system for rendering, completely decoupled from the game's way of rendering things. This made the task a pretty big refactor, since I wanted both instances to use the same render pipeline. With the removal of the old editor render passes, I built the same render commands out of the editor's representation of objects (we had different representations in game and in the editor). The editor also had an ID pass which I had to fit into the render pipeline somehow. Finally, I managed to merge the two, making it much easier to add new render features.

// Example of building a render command in the Editor:
for (const auto& property : properties)
{
    // ***** REGULAR MODELS *****
    if (property.type == Tga::GetPropertyType<Tga::CopyOnWriteWrapper<Tga::SceneModel>>())
    {
        const Tga::SceneModel& sceneModel = property.value.Get<Tga::CopyOnWriteWrapper<Tga::SceneModel>>()->Get();
        std::shared_ptr<Tga::Model> model = aDrawParams.cache.GetModelUsingCache(sceneModel.path);
        if (!model) { continue; }
        const size_t meshCount = model->GetMeshCount();
        if (meshCount == 0) { continue; }
        Tga::Vector3f scale = aSceneObject.GetScale();
        float maxScale = std::max(scale.x, std::max(scale.y, scale.z));
        if (!CheckBounds(aDrawParams.frustum, aSceneObject.GetTransform(), maxScale, *model))
            continue;
        commands.reserve(commands.size() + meshCount);
        const Tga::ModelShader* shader = aDrawParams.viewport.GetOrCreateShader(sceneModel);
        if (!shader) { continue; }
        Tga::LayerType objectLayer = Tga::LayerType::Default;
        for (const auto& prop : properties)
        {
            if (prop.type == Tga::GetPropertyType<Tga::CopyOnWriteWrapper<Tga::Layer>>())
            {
                const Tga::Layer& layer = prop.value.Get<Tga::CopyOnWriteWrapper<Tga::Layer>>()->Get();
                objectLayer = layer.type;
                break;
            }
        }
        // Build command for each mesh
        for (size_t i = 0; i < meshCount && i < MAX_MESHES_PER_MODEL; ++i)
        {
            RenderCommand cmd = {};
            cmd.type          = RenderCommandType::Mesh;
            cmd.model         = model.get();
            cmd.meshIndex     = static_cast<int>(i);
            cmd.transform     = transform;
            cmd.shader        = shader;
            cmd.boneTransforms = nullptr;
            cmd.rasterizerState   = sceneModel.rasterizerState;
            cmd.blendState        = sceneModel.blendState;
            cmd.depthStencilState = Tga::DepthStencilState::WriteLessOrEqual;
            cmd.castsShadows  = sceneModel.castsShadow;
            cmd.layer         = objectLayer;
            cmd.shouldFade    = false;
            for (int j = 0; j < 4; ++j)
            {
                if (!sceneModel.textures[i][j].IsEmpty())
                {
                    Tga::TextureSrgbMode srgbMode = (j == 0) ? Tga::TextureSrgbMode::ForceSrgbFormat
                        : Tga::TextureSrgbMode::ForceNoSrgbFormat;
                    cmd.textures[j] = aDrawParams.cache.GetTextureUsingCache(sceneModel.textures[i][j], srgbMode);
                }
                else
                {
                    cmd.textures[j] = model->GetDefaultTextures(i)[j];
                }
            }
            commands.push_back(cmd);
        }
    }
}
Single Asset Preview & VFX Live Editing

A long-standing request from our first game was to preview VFXes and single assets with correct lighting. So I added the ability to preview individual assets in the editor with live lighting, and to preview and live-edit VFX while viewing them. The artist could for example change constant buffer parameters to see their changes apply live.

void ObjectDefinitionDocument::Update(float aTimeDelta, InputManager& inputManager)
{
    inputManager;
    // ***** CLEAR CACHE *****
    myCache.ClearCache(myPreviewIsAnimated);
    // Mainly checks if it is a VFX and updates it accordingly,
    // but also clears the spawned object if it is not a VFX
    UpdatePreviewObject();
    // Update VFX object
    if (myPreviewIsVFX && myPreviewVFX)
    {
        myPreviewVFX->Update(aTimeDelta);
        if (myPreviewVFX->GetComponent<VFXInstance>()->IsExpired() &&
            myPreviewVFX->GetComponent<VFXInstance>()->ShouldLoop())
        {
            myPreviewVFX->GetComponent<VFXInstance>()->SetSpawnedTime(
                Tga::Engine::GetInstance()->GetTotalTime());
        }
    }
}
void ObjectDefinitionDocument::UpdatePreviewObject()
{
    myPreviewIsVFX      = false;
    myPreviewIsAnimated = false;
    myPreviewIsParticle = false;
    const auto properties = myObjectDefinition->GetProperties();
    for (const auto& property : properties)
    {
        if (property.type == Tga::GetPropertyType<Tga::CopyOnWriteWrapper<Tga::TagData>>())
        {
            const Tga::TagData& tag = property.value.Get<Tga::CopyOnWriteWrapper<Tga::TagData>>()->Get();
            if (tag.tag == Tga::Tag::VFX)      { myPreviewIsVFX = true; }
            if (tag.tag == Tga::Tag::Particle)  { myPreviewIsParticle = true; }
        }
        if (property.type == Tga::GetPropertyType<Tga::CopyOnWriteWrapper<Tga::SceneAnimatedModel>>())
        {
            myPreviewIsAnimated = true;
        }
        if (property.type == Tga::GetPropertyType<Tga::CopyOnWriteWrapper<ParticleEmitterData>>())
        {
            const ParticleEmitterData& rootData =
                property.value.Get<Tga::CopyOnWriteWrapper<ParticleEmitterData>>()->Get();
            size_t needed = 1 + rootData.subEmitters.size();
            if (myEmitters.size() != needed)
            {
                myEmitters.clear();
                myEmitters.resize(needed);
            }
            myEmitters[0].data = rootData;
            for (size_t i = 0; i < rootData.subEmitters.size(); i++)
                myEmitters[i + 1].data = rootData.subEmitters[i];
        }
    }
    if (!myPreviewIsParticle) { myPreviewParticles.clear(); myEmitters.clear(); }
    if (!myPreviewIsVFX)
    {
        myPreviewVFX.reset();
        myLastBuiltDefinition = nullptr;
        return;
    }
    // Build or update the properties depending on if we open a new tgo or not
    if (!myPreviewVFX || myLastBuiltDefinition != myObjectDefinition)
    {
        myPreviewVFX = std::make_shared<GameObject>();
        BuildSpawnedObjectFromProperties(properties, myCache, myViewport,
            myViewport.GetRenderManager(), myPreviewVFX);
        myLastBuiltDefinition = myObjectDefinition;
    }
    else
    {
        UpdateSpawnedObjectFromProperties(myPreviewVFX->GetComponent<VFXInstance>(),
            properties, myCache, myViewport, myViewport.GetRenderManager());
    }
}
Enemy Behaviour

Integrated enemy behaviour using third-party libraries for navmesh generation, pathfinding and agent separation. With limited time, since I hopped on the enemy task force pretty late, I reused the state machine as a behaviour tree since that was one of the requirements for this project. Each state represented a node in the tree, with different phases such as Enter/Exit/Update. I also implemented an alert system, with a ripple effect, to create a creepy spider-like reaction for the enemies.

class EnemyAlertSystem
{
public:
    static EnemyAlertSystem& Get()
    {
        static EnemyAlertSystem instance;
        return instance;
    }
    EnemyAlertSystem(const EnemyAlertSystem&) = delete;
    void operator=(const EnemyAlertSystem&) = delete;
    void RegisterEnemy(const std::shared_ptr<Enemy>& anEnemy);
    void UnregisterEnemy(const std::shared_ptr<Enemy>& anEnemy);
    void BroadcastAlert(const Tga::Vector3f& aSourcePos, float aRadius, const std::shared_ptr<Enemy>& aSourceEnemy);
private:
    EnemyAlertSystem() = default;
    std::vector<std::weak_ptr<Enemy>> myEnemies;
};

void EnemyAlertSystem::BroadcastAlert(const Tga::Vector3f& aSourcePos, float aRadius, const std::shared_ptr<Enemy>& aSourceEnemy)
{
    const float radiusSq = aRadius * aRadius;
    // Clean up expired weak_ptrs while iterating
    myEnemies.erase(std::ranges::remove_if(myEnemies,
        [](const std::weak_ptr<Enemy>& aWeak) { return aWeak.expired(); }).begin(),
        myEnemies.end());
    for (auto& weak : myEnemies)
    {
        auto enemy = weak.lock();
        if (!enemy || enemy == aSourceEnemy) { continue; }
        if ((enemy->GetPosition() - aSourcePos).LengthSqr() <= radiusSq)
        {
            enemy->ReceiveAlert(aSourcePos);
        }
    }
}

void HumanBrainTree::OnAlert(const Tga::Vector3f& aSourcePos)
{
    if (myContext.enemy->GetData().isPassive) { return; }
    if (myContext.enemyState == EnemyStateID::Engage ||
        myContext.enemyState == EnemyStateID::Attack) { return; }
    const float distance = (myContext.enemy->GetPosition() - aSourcePos).Length();
    myAlertDelay = distance / 500.f;
    myIsAlerted = true;
    myContext.enemyState = EnemyStateID::Engage;
    AudioManager::GetInstance()->PlaySpatialisedAudio(
        AudioConstExpr::ENEMY_3D_Engage, myContext.enemy->GetTransform());
}
